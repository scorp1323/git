1 - mkdir site_1 -  утв. нової теки site_1
2 - cd site_1 - перейти в теку site_1
3 - git init - ініціювати директорію, в якій знах.
4 - ls - показати всі файли та папки в даній текі
4 - ls -a - показати всі файли + сховані та папки в даній текі
4 - ls -l - показати всі файли у вигляді таб. з правами

              - РОБОТА З GIT - 
1 - git status - проверка состояния
2 - git log - показати всі коміти
3 - git add <file1> <file2> - подготавливаем коммит, добавляя в него файлы
4 - git add . - подготавливаем коммит, добавляя в него  все файлы
5 - git rm <file1> <file2>  - удаляем устаревшие файлы
6 - git commit -m 'Комментарий к коммиту' - выполняем коммит (слепок)
-----------------------------------

2 - git checkout main - перейти на вітку main

              - УДАЛЕННЫЙ КОММИТ -
1 - git remote add origin https://github.com/scorp1323/site1.git - подключится к удаленному репозиторию и дать ему имя origin

2 - git push -u origin master - загрузка данных с тек. дир. на удаленный репозиторий 

3 - git pull origin master - автомат. извлекает и затем сливает данные из удалённой ветки в вашу текущую ветку

4 - git remote - перечисляет список имён-сокращений для всех уже указанных удалённых дескрипторов.

5 - git remote -v    - посмотреть, какому URL соответствует сокщ. имя в Git

6 - git clone https://github.com/scorp1323/site1.git - скопировать содержимое с удал. репоз. в тек. дир.

7 - git remote set-url origin https://github.com/scorp1323/new.git - меняет адресс удаленного репозитория origin на данный (на https://github.com/scorp1323/new.git)

8 - git pull origin master --rebase - Онови локальний репозиторій якщо твоя локальна гілка відстає від віддаленого репозиторію

9 - git rebase --continue - Якщо були конфлікти при злитті, виправ їх, збережи файли та виконай:
 - git add .
 - git rebase --continue
